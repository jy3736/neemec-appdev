from firebase import firebase
from random import randrange, random
from time import sleep
import sys

db_url = 'https://你的資料庫網址.firebaseio.com/'
fdb = firebase.FirebaseApplication(db_url, None)

T1, T2 = 25, 25

def sample():
    global T1, T2
    T1 = int((T1+random()*randrange(-1,2))*10)/10    
    T2 = int((T2+random()*randrange(-1,2))*10)/10    
    print(f"T1 = {T1:4}  T2 = {T2:4}")   

def main(sr=1, cd=120):
    global T1, T2
    cntdown = cd
    while(cntdown>0):
        cntdown-=1
        sample()
        fdb.put('/Meter', 'T1', T1)
        fdb.put('/Meter', 'T2', T2)
        sleep(sr)

if __name__=="__main__":
    main()
